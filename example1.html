Marker Clustering
OverView

This tutorial shows you how to use marker clusters to display a large number of markers on a map. You can use the MarkerClusterer library in combination with the Maps JavaScript API to combine markers of close proximity into clusters, and simplify the display of markers on the map. To see how marker clustering works, view the map below.

    function initMap() {
      var uluru = {lat: 24.911867817255185, lng: 121.65543315726359};
      var map = new google.maps.Map(document.getElementById('map'), {
        zoom: 4,
        center: uluru
      });
      var marker = new google.maps.Marker({
        position: uluru,
        map: map
      });

      var infowindow = null;

      google.maps.event.addListener(map, "click", function (e) {
        if (infowindow) {
            infowindow.close();
        }
        marker.setPosition({lat: e.latLng.lat(), lng: e.latLng.lng()});
      });

      google.maps.event.addListener(marker, 'click', function(e) {
        if (infowindow) {
            infowindow.close();
        }
        contentString = '<div id="content">lat: '+ e.latLng.lat() +'<br>lng' + e.latLng.lng() + '</div>';
        infowindow = new google.maps.InfoWindow({
          content: contentString
        });
        infowindow.open(map, marker);
      });
    }
